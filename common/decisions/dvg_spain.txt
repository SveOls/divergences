integrate_granada = {
	is_shown = {
		OR = {
			AND = {
				exists = c:SPC
				this = c:SPC
			}
			AND = {
				exists = c:SPA
				this = c:SPA
			}
		}
		exists = c:GRA
		c:GRA = {
			OR = {
				is_subject_of = c:SPA
				is_subject_of = c:SPC
			}
		}
		NOT = { has_global_variable = granada_annexed }
	}
	possible = {
		country_rank = rank_value:great_power
		has_technology_researched = pan-nationalism
		c:GRA = {
			is_subject = yes
			liberty_desire <=20
		}
		
		at_war = no
	}
	when_taken = {
		annex_with_incorporation = c:GRA
		add_primary_culture = cu:granadan
		set_global_variable = granada_annexed
	}
}

end_the_andine_wars = {
	is_shown = {
		OR = {
			AND = {
				exists = c:SPC
				this = c:SPC
			}
			AND = {
				exists = c:SPA
				this = c:SPA
			}
		}
		exists = c:TWT
		c:QUI = { is_subject_of = c:SPA }
		NOT = { has_global_variable = andine_war_declared }
	}
	possible = {
		country_rank = rank_value:great_power
		has_technology_researched = civilizing_mission
		at_war = no
		NOT = { has_truce_with = c:TWT }
	}
	when_taken = {
		trigger_event = { id = dvg_spain.107 }
		set_global_variable = andine_war_declared
	}
